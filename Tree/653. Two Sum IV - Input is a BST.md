from collections import deque
class Solution:
    def findTarget(self, root, k):
        """
        :type root: TreeNode
        :type k: int
        :rtype: bool
        """
        if not root:
            return False
        queue = deque()
        queue.append(root)
        visited = set()
        while queue:
            cur = queue.popleft()
            if k - cur.val in visited:
                return True
            else:
                visited.add(cur.val)
            if cur.left:
                queue.append(cur.left)
            if cur.right:
                queue.append(cur.right)

        return False

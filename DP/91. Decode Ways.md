class Solution:
    def numDecodings(self, s):
        """
        :type s: str
        :rtype: int
        """
        size = len(s)
        
        if size ==0:
            return 0
        elif size ==1 and s[0] !='0':
            return 1
        #dp[i] 长度为i的消息，可以被解析到可能性
        dp=[0]* (size+1)

        if s[0] =='0':
            return 0
        else:
            dp[1]=1  #一个数字，  一种解析

        if int(s[0:2]) <=26 and s[1]!='0':
            dp[2] =2
        elif int(s[0:2]) %10==0 and int(s[0])>2:   # 30 40 50 60 70 80 90 不难解析
            dp[2]=0
        else:
            dp[2] =1 
           


        #function
        for i in range(3,size+1):

            if int(s[i-1]) ==0: # 末尾为0 的情况
                if 1<=int(s[i-2]) <=2: #倒数第二位为1 或2 才有解
                    dp[i]=dp[i-2]
                else:
                    dp[i]=0
            elif int(s[i-1]) <=6:  #  末尾1-6
                if 1<=int(s[i-2])<=2: # 倒数第二位 为1 和2 
                    dp[i]= dp[i-1]+dp[i-2]
                else:
                    dp[i]=dp[i-1]
            else:  #末尾7-9
                if s[i-2] =='1':   # 倒数第二位 为1
                    dp[i]= dp[i-1]+dp[i-2]
                else:
                    dp[i]=dp[i-1]
                 


        return dp[size]

# -*- coding: UTF-8 -*-
import collections
class Solution:
    def exist(self, board, word):
        """
        :type board: List[List[str]]
        :type word: str
        :rtype: bool
        """
        if not word:
            return True

        row = len(board)
        col = len(board[0])
        visited = set()
        for i in range(row):
            for j in range(col):
                if board[i][j] == word[0]:
                    visited.add((i,j))
                    if self.backtacking(board,word[1:],i,j,row,col,visited):
                        return True
                    visited.remove((i,j))
        
        return False

    def backtacking(self,board,word,x,y,row,col,visited):

        dx = [0,0,1,-1]
        dy = [1,-1,0,0]
        if word == "":
            return True

        flag = False
        for i in range(4):
            newx = x + dx[i]
            newy = y + dy[i]
            if 0 <= newx < row and 0 <= newy <col:
                if word[0] == board[newx][newy] and (newx,newy) not in visited:
                    visited.add((newx,newy))
                    flag = self.backtacking(board,word[1:],newx,newy,row,col,visited)
                    if flag :
                        return True 
                    visited.remove((newx,newy))
        return flag

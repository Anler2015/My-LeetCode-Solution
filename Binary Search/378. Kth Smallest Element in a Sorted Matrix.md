class Solution {
    public int kthSmallest(int[][] matrix, int k) {
        if(matrix.length <= 0 || matrix[0].length <= 0){
            return 0;
        }
        int n = matrix.length;
        int start = matrix[0][0];
        int end = matrix[n-1][n-1];

        while(start<end){
            int mid = (start + end)/2;
            int count = 0, j = n-1;
            for(int i = 0;i<n;i++){
                while(j>=0 && matrix[i][j]>mid){
                    j--;
                }
                count += (j+1);
            }
            
            if(count<k){
                start = mid+1;
            }else{
                end = mid;
            }

        }

        return start;
    }
}

class Solution(object):
    def isValidBST(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        if not root:
            return True
        res=self.inorder(root)
        for i in range(0,len(res)-1):
            if res[i] >= res[i+1]:
                return False

        return True


    def inorder(self,root):
        if not root:
            return []

        left = self.inorder(root.left)
        right= self.inorder(root.right)
        temp=[]
        temp.extend(left)
        temp.append(root.val)
        temp.extend(right)
        return temp

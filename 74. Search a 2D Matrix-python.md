class Solution:
    def searchMatrix(self, matrix, target):
        """
        :type matrix: List[List[int]]
        :type target: int
        :rtype: bool
        """
        if not matrix or not matrix[0]:
            return False

        if target < matrix[0][0]:
            return False

        row = len(matrix)-1
        col = len(matrix[0])-1
        

        start, end = 0, row

        while start < end:
            mid = (start+end)//2
            if matrix[mid][0] == target:          #每个矩阵首元素， 判断区间
                return True

            elif matrix[mid][0] > target:
                 end = mid-1 
            else:
                 start = mid+1
    
        if matrix[start][0] < target:      #找出第几行，也就是区间
            targetline =start
        elif matrix[start][0]>target:
            targetline = start-1
        else:
            return True
                    
        start,end = 0, col

        while start <= end:       # 取等号的原因是因为， 当start=end 的时候也要检查该元素
            mid = (start+end)//2
            if matrix[targetline][mid] == target:          
                return True

            elif matrix[targetline][mid] > target:
                 end = mid-1  
            else:
                 start = mid+1

        return False

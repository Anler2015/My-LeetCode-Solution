from collections import deque

class Solution:
    def ladderLength(self, beginWord, endWord, wordList):
        """
        :type beginWord: str
        :type endWord: str
        :type wordList: List[str]
        :rtype: int
        """
        if not wordList:
            return 0

        if wordList.count(endWord) == 0:
            return 0
     
        wordList.append(beginWord)  # 第一个词入wordlist
        graph = [ set() for i in range(len(wordList))]
        visited = [False] * len(wordList)
        count = 0

        wordLength = len(wordList[0])
        #构图   无向无权图
        for i in range(len(wordList)):
            for j in range(i+1,len(wordList)):
                cur = wordList[i]
                sub = wordList[j]
                diff = 0
                for k in range(wordLength):
                    if cur[k] != sub[k]:
                        diff +=1
                if diff == 1:
                    graph[i].add(sub)
                    graph[j].add(cur)


        queue = deque()
        queue.append(beginWord)

        while len(queue) != 0:
            size = len(queue)
            count +=1       
            
            for i in range(size):     # 每层计数一次
                curWord = queue.popleft()

                if curWord == endWord:
                    return count

                index = wordList.index(curWord) 
                graphNode = graph[index]
                while len(graphNode) != 0:
                    temp = graphNode.pop()
                    if visited[wordList.index(temp) ] == False:
                        queue.append(temp)
                        visited[wordList.index(temp)] = True

        return 0

from collections import deque

class Solution:
    def ladderLength(self, beginWord, endWord, wordList):
        """
        :type beginWord: str
        :type endWord: str
        :type wordList: List[str]
        :rtype: int
        """
        if not wordList:
            return 0

        if wordList.count(endWord) == 0:
            return 0

     

        wordLength = len(wordList[0])
        wordList = set(wordList)

        queue = deque()
        queue.append([beginWord,1])

        while queue:
            curWord,count = queue.popleft()
            if curWord == endWord:
                return count
            for i in range(wordLength):
                for c in "abcdefghijklmnopqrstuvwxyz":
                    newWord = curWord[:i]+c+curWord[i+1:]
                    if newWord in wordList:
                        queue.append([newWord,count+1])
                        wordList.remove(newWord)
    
        return 0

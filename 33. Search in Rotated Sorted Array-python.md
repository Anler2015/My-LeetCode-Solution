class Solution:
	def search(self, nums, target):

		if nums== None or len(nums)==0:
			return -1
		start =0
		end = len(nums)-1
		while start<end:
			mid = int((start+end)/2)
			if nums[mid] >= nums[end]:
				start = mid+1;
			else:
				end = mid

		line=start
		if line ==0:
				start = 0
				end = len(nums)-1
				while start<=end:
					mid = int((start+end)/2)
					if nums[mid] > target:
						end = mid-1
					elif nums[mid]< target:
						start = mid+1
					else:
						return mid
		if target>= nums[line] and target <= nums[len(nums)-1]: 
				start = line
				end = len(nums)-1
				while start<=end:
					mid = int((start+end)/2)
					if nums[mid] > target:
						end = mid-1
					elif nums[mid]< target:
						start = mid+1
					else:
						return mid
		elif target>= nums[0] and target<= nums[line-1]:
				start = 0
				end = line-1
				while start<=end:
					mid = int((start+end)/2)
					if nums[mid] > target:
						end = mid-1
					elif nums[mid]< target:
						start = mid+1
					else:
						return mid

		return -1

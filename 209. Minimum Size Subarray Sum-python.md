双重循环 时间复杂度n*n   超时
```
class Solution:
    def minSubArrayLen(self, s, nums):
        """
        :type s: int
        :type nums: List[int]
        :rtype: int
        """
        start = 0
        minLen = 0
        first= True
        for n in range(len(nums)):
            rest=s-nums[n]
            if rest <=0:
                return 1
            else:
                start = n
            for i in range(n+1,len(nums)):
                rest=rest-nums[i]
                if rest <=0:
                    if first==True:
                        minLen = i-start+1
                        first = False
                    if i-start+1 <minLen:
                        minLen = i-start+1
                    break
               
        return minLen
```

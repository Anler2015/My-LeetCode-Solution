# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def rotateRight(self, head, k):
        """
        :type head: ListNode
        :type k: int
        :rtype: ListNode
        """
        if not head:
            return
        for i in range(k):
            head = self.rotate(head)

        return head
        

    def rotate(self,head):
        last = head
        prev = None
        while last.next != None:
            prev = last
            last = last.next
            
        if prev != None:  # 防止链表长度为1的情况
            last.next = head
            prev.next = None

        return last

    def rotateRight(self, head, k):
        """
        :type head: ListNode
        :type k: int
        :rtype: ListNode
        """
        if not head:
            return
        
        count_head = head
        count = 0
        while  count_head:
            count+=1
            count_head = count_head.next
            
        rotata_size = k %count
        for i in range(rotata_size):
            head = self.rotate(head)

        return head
        

    def rotate(self,head):
        last = head

        while last.next != None:
            prev = last
            last = last.next
            
        if prev != None:  # 防止链表长度为1的情况
